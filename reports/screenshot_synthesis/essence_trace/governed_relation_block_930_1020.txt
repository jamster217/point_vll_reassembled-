    def _naturalize_seed_clause(t: str) -> str:
        base = _clean_line(t or "")
        if not base:
            return ""
        base = re.sub(r"\s+while\s+", " ", base, flags=re.IGNORECASE)
        base = re.sub(r"\s+", " ", base).strip()
        if base:
            base = base[0].upper() + base[1:]
        if not re.search(r"[.!?…]$", base):
            base += "."
        return base
